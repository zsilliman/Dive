179 SDL_FORCE_INLINE float marshall(float value) {
180 #if SDL_BYTEORDER == SDL_LIL_ENDIAN
181  union
182  {
183  float f;
184  Uint32 ui32;
185  } swapper;
186  swapper.f = value;
187  swapper.ui32 = SDL_Swap32(swapper.ui32);
188  return swapper.f;
189 #else
190  return value;
191 #endif
